$env:JAVA_HOME = "C:\Program Files\Eclipse Adoptium\jdk-21.0.11.10-hotspot"

$env:Path = "$env:JAVA_HOME\bin;" + (($env:Path -split ';' | Where-Object {
$_ -and ($_ -notmatch 'jdk1\.8|jdk-1\.8|jre1\.8|Java\\javapath|Oracle\\Java')
}) -join ';')

where.exe java
java -version
mvn -v

mvn clean package -DskipTests

cd F:\IdeaWorkSpace\Myself\astock-emotion-system

$projectRoot = "F:\IdeaWorkSpace\Myself\astock-emotion-system"
$projectName = "astock-emotion-system"
$zipFile = "F:\IdeaWorkSpace\Myself\astock-emotion-system.zip"

if (Test-Path $zipFile) {
Remove-Item $zipFile -Force
}

Add-Type -AssemblyName System.IO.Compression.FileSystem

$excludeDirs = @(
"\.git\",
"\.idea\",
"\node_modules\",
"\dist\",
"\target\",
"\out\",
"\build\"
)

$excludeFiles = @(
".env",
".DS_Store"
)

$zip = [System.IO.Compression.ZipFile]::Open($zipFile, [System.IO.Compression.ZipArchiveMode]::Create)

try {
Get-ChildItem -Path $projectRoot -Recurse -File -Force | ForEach-Object {
$fullPath = $_.FullName
$relativePath = $fullPath.Substring($projectRoot.Length).TrimStart('\')

        $normalized = "\" + $relativePath.Replace("/", "\")
        $fileName = $_.Name

        $skipDir = $false
        foreach ($dir in $excludeDirs) {
            if ($normalized -like "*$dir*") {
                $skipDir = $true
                break
            }
        }

        if ($skipDir) {
            return
        }

        if ($excludeFiles -contains $fileName) {
            return
        }

        if ($fileName -like "*.class" -or $fileName -like "*.log") {
            return
        }

        $entryName = "$projectName/" + $relativePath.Replace("\", "/")

        [System.IO.Compression.ZipFileExtensions]::CreateEntryFromFile(
            $zip,
            $fullPath,
            $entryName,
            [System.IO.Compression.CompressionLevel]::Optimal
        ) | Out-Null
    }
}
finally {
$zip.Dispose()
}

Get-Item $zipFile
