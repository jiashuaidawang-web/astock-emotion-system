[//]: # (后端部署一键脚本)
set -e

APP_HOME=/opt
APP_NAME=astock-emotion-system
APP_DIR=${APP_HOME}/${APP_NAME}
PACKAGE=${APP_HOME}/astock-emotion-system.zip
BAK_DIR=${APP_HOME}/${APP_NAME}_bak_$(date +%Y%m%d_%H%M%S)

cd ${APP_HOME}

echo "1. 检查部署包"
ls -lh ${PACKAGE}

echo "2. 停止旧 frontend/backend，不停止 mysql/clickhouse"
if [ -d "${APP_DIR}" ]; then
docker compose -f ${APP_DIR}/docker-compose.yml stop backend frontend || true
fi

echo "3. 备份旧工程目录"
if [ -d "${APP_DIR}" ]; then
mv ${APP_DIR} ${BAK_DIR}
echo "旧目录已备份到: ${BAK_DIR}"
fi

echo "4. 解压新工程"
unzip -q ${PACKAGE} -d ${APP_HOME}

echo "5. 确认新工程目录"
cd ${APP_DIR}
pwd
ls -la

echo "6. 如果旧目录有 .env，则恢复旧 .env"
if [ -f "${BAK_DIR}/.env" ]; then
cp ${BAK_DIR}/.env ${APP_DIR}/.env
echo ".env 已从旧目录恢复"
else
echo "旧目录没有 .env，请确认新包里是否自带 .env"
fi

echo "7. 禁用有问题的 docker-compose.override.yml"
if [ -f docker-compose.override.yml ]; then
mv docker-compose.override.yml docker-compose.override.yml.bak.$(date +%Y%m%d_%H%M%S)
echo "docker-compose.override.yml 已备份停用"
fi

echo "8. 写入根目录 .dockerignore，避免 node_modules/target 被复制进镜像"
cat > .dockerignore <<'EOF'
.git
.idea
.vscode

**/node_modules
**/dist
**/target
**/.DS_Store

*.log
*.tmp
*.zip
*.tar
*.tar.gz

docker-compose.override.yml
EOF

echo "9. 清理宿主机前端依赖和构建产物"
rm -rf astock-frontend/node_modules
rm -rf astock-frontend/dist

echo "10. 验证 compose 配置"
docker compose -f docker-compose.yml config >/tmp/astock-compose-check.yml
echo "compose 配置 OK"

echo "11. 构建 backend/frontend"
docker compose -f docker-compose.yml build --no-cache backend frontend

echo "12. 启动数据库"
docker compose -f docker-compose.yml up -d mysql clickhouse

echo "13. 启动 backend/frontend"
docker compose -f docker-compose.yml up -d backend frontend

echo "14. 查看容器状态"
docker compose -f docker-compose.yml ps

echo "15. 检查后端健康"
sleep 10
curl -i http://127.0.0.1:${BACKEND_PORT:-8080}/actuator/health || true

echo "部署完成"
